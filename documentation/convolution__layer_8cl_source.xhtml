105 __kernel void im2col_generic(
106  TENSOR3D_DECLARATION(src),
107  IMAGE_DECLARATION(dst),
108  int kernel_size,
109  int kernel_depth,
110  int width,
111  int2 input_dims,
112  int2 strides,
113  int2 paddings)
114 {
115  Tensor3D src = CONVERT_TO_TENSOR3D_STRUCT(src);
116  Image dst = CONVERT_TO_IMAGE_STRUCT_NO_STEP(dst);
117 
118  // Determine output index
119  uint idx = (get_global_id(1) * width + get_global_id(0)) * dst.stride_y;
120  __global uchar *output_ptr = dst.ptr + idx;
121 
122  // Determine current input index
123  const int top_left_x = get_global_id(0) * strides.x - paddings.x;
124  const int top_left_y = get_global_id(1) * strides.y - paddings.y;
125 
126  // Linearize convolution elements
127  for(int d = 0; d < kernel_depth; ++d)
128  {
129  for(int y = top_left_y, y_e = top_left_y + kernel_size; y < y_e; ++y)
130  {
131  for(int x = top_left_x, x_e = top_left_x + kernel_size; x < x_e; ++x, output_ptr += dst.stride_x)
132  {
133  if(x < 0 || x >= input_dims.x || y < 0 || y >= input_dims.y)
134  {
135  *((__global DATA_TYPE *)output_ptr) = 0;
136  }
137  else
138  {
139  *((__global DATA_TYPE *)output_ptr) = *((__global DATA_TYPE *)(tensor3D_offset(&src, x, y, d)));
140  }
141  }
142  }
143  }
144 
145 #if defined HAS_BIAS
146  *((__global DATA_TYPE *)output_ptr) = 1;
147 #endif
148 }
