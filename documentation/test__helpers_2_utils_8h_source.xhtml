139  template <typename T>
140  void fill_image(T &image)
141  {
142  ARM_COMPUTE_ERROR_ON(!is_open());
143  ARM_COMPUTE_ERROR_ON(image.info()->dimension(0) != _width || image.info()->dimension(1) != _height);
144  ARM_COMPUTE_ERROR_ON_FORMAT_NOT_IN(&image, arm_compute::Format::U8, arm_compute::Format::RGB888);
145  try
146  {
147 #ifdef ARM_COMPUTE_CL
148  // Map buffer if creating a CLTensor
149  if(std::is_same<typename std::decay<T>::type, arm_compute::CLImage>::value)
150  {
151  image.map();
152  }
153 #endif
154  // Check if the file is large enough to fill the image
155  const size_t current_position = _fs.tellg();
156  _fs.seekg(0, std::ios_base::end);
157  const size_t end_position = _fs.tellg();
158  _fs.seekg(current_position, std::ios_base::beg);
159 
160  ARM_COMPUTE_ERROR_ON_MSG((end_position - current_position) < image.info()->tensor_shape().total_size() * image.info()->element_size(),
161  "Not enough data in file");
162  ARM_COMPUTE_UNUSED(end_position);
163 
164  switch(image.info()->format())
165  {
166  case arm_compute::Format::U8:
167  {
168  // We need to convert the data from RGB to grayscale:
169  // Iterate through every pixel of the image
170  arm_compute::Window window;
171  window.set(arm_compute::Window::DimX, arm_compute::Window::Dimension(0, _width, 1));
172  window.set(arm_compute::Window::DimY, arm_compute::Window::Dimension(0, _height, 1));
173 
174  arm_compute::Iterator out(&image, window);
175 
176  unsigned char red = 0;
177  unsigned char green = 0;
178  unsigned char blue = 0;
179 
180  arm_compute::execute_window_loop(window, [&](const arm_compute::Coordinates & id)
181  {
182  red = _fs.get();
183  green = _fs.get();
184  blue = _fs.get();
185 
186  *out.ptr() = 0.2126f * red + 0.7152f * green + 0.0722f * blue;
187  },
188  out);
189 
190  break;
191  }
192  case arm_compute::Format::RGB888:
193  {
194  // There is no format conversion needed: we can simply copy the content of the input file to the image one row at the time.
195  // Create a vertical window to iterate through the image's rows:
196  arm_compute::Window window;
197  window.set(arm_compute::Window::DimY, arm_compute::Window::Dimension(0, _height, 1));
198 
199  arm_compute::Iterator out(&image, window);
200 
201  arm_compute::execute_window_loop(window, [&](const arm_compute::Coordinates & id)
202  {
203  // Copy one row from the input file to the current row of the image:
204  _fs.read(reinterpret_cast<std::fstream::char_type *>(out.ptr()), _width * image.info()->element_size());
205  },
206  out);
207 
208  break;
209  }
210  default:
211  ARM_COMPUTE_ERROR("Unsupported format");
212  }
213 
214 #ifdef ARM_COMPUTE_CL
215  // Unmap buffer if creating a CLTensor
216  if(std::is_same<typename std::decay<T>::type, arm_compute::CLTensor>::value)
217  {
218  image.unmap();
219  }
220 #endif
221  }
222  catch(const std::ifstream::failure &e)
223  {
224  ARM_COMPUTE_ERROR("Loading PPM file: %s", e.what());
225  }
226  }
227 
228 private:
229  std::ifstream _fs;
230  unsigned int _width, _height;
231 };

242 template <typename T>
243 void save_to_ppm(T &tensor, const std::string &ppm_filename)
244 {
245  ARM_COMPUTE_ERROR_ON_FORMAT_NOT_IN(&tensor, arm_compute::Format::RGB888, arm_compute::Format::U8);
246  ARM_COMPUTE_ERROR_ON(tensor.info()->num_dimensions() > 2);
247 
248  std::ofstream fs;
249 
250  try
251  {
252  fs.exceptions(std::ofstream::failbit | std::ofstream::badbit | std::ofstream::eofbit);
253  fs.open(ppm_filename, std::ios::out | std::ios::binary);
254 
255  const unsigned int width = tensor.info()->tensor_shape()[0];
256  const unsigned int height = tensor.info()->tensor_shape()[1];
257 
258  fs << "P6\n"
259  << width << " " << height << " 255\n";
260 
261 #ifdef ARM_COMPUTE_CL
262  // Map buffer if creating a CLTensor
263  if(std::is_same<typename std::decay<T>::type, arm_compute::CLTensor>::value)
264  {
265  tensor.map();
266  }
267 #endif
268 
269  switch(tensor.info()->format())
270  {
271  case arm_compute::Format::U8:
272  {
273  arm_compute::Window window;
274  window.set(arm_compute::Window::DimX, arm_compute::Window::Dimension(0, width, 1));
275  window.set(arm_compute::Window::DimY, arm_compute::Window::Dimension(0, height, 1));
276 
277  arm_compute::Iterator in(&tensor, window);
278 
279  arm_compute::execute_window_loop(window, [&](const arm_compute::Coordinates & id)
280  {
281  const unsigned char value = *in.ptr();
282 
283  fs << value << value << value;
284  },
285  in);
286 
287  break;
288  }
289  case arm_compute::Format::RGB888:
290  {
291  arm_compute::Window window;
292  window.set(arm_compute::Window::DimX, arm_compute::Window::Dimension(0, width, width));
293  window.set(arm_compute::Window::DimY, arm_compute::Window::Dimension(0, height, 1));
294 
295  arm_compute::Iterator in(&tensor, window);
296 
297  arm_compute::execute_window_loop(window, [&](const arm_compute::Coordinates & id)
298  {
299  fs.write(reinterpret_cast<std::fstream::char_type *>(in.ptr()), width * tensor.info()->element_size());
300  },
301  in);
302 
303  break;
304  }
305  default:
306  ARM_COMPUTE_ERROR("Unsupported format");
307  }
308 #ifdef ARM_COMPUTE_CL
309  // Unmap buffer if creating a CLTensor
310  if(std::is_same<typename std::decay<T>::type, arm_compute::CLTensor>::value)
311  {
312  tensor.unmap();
313  }
314 #endif
315  }
316  catch(const std::ofstream::failure &e)
317  {
318  ARM_COMPUTE_ERROR("Writing %s: (%s)", ppm_filename.c_str(), e.what());
319  }
320 }
321 }
